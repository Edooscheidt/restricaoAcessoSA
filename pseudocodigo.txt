ALGORITMO: Controle de Acesso Corporativo - Block&Go
SISTEMA: Fluxograma SA (Restrição de Acesso)

INÍCIO DO SISTEMA

    // --- Etapa de Inicialização do Hardware ---
    INICIALIZAR biblioteca SPI
    INICIALIZAR leitor RFID (MFRC522)
    INICIALIZAR comunicação serial (Hardware UART)

    // --- Definição de Variáveis Globais de Segurança ---
    DEFINIR tag_autorizada
    DEFINIR status_funcionario <- "Ativo"
    DEFINIR permissao_acesso   <- "Autorizado"

    EXIBIR "Block&Go: Sistema iniciado com sucesso."
    EXIBIR "Status: Aguardando aproximação da tag RFID..."

    // --- Loop Principal do Ecossistema ---
    ENQUANTO (sistema_ligado == VERDADEIRO) FAÇA

        SE (nova_tag_detectada) ENTÃO
            LER tag_usuario

            // [VALIDAÇÃO 1] Verificação de autenticidade física da Tag
            SE (tag_usuario != tag_autorizada) ENTÃO
                EXIBIR "ALERTA: Acesso negado. Tag inválida ou não cadastrada."
                REGISTRAR_LOG(tentativa_acesso, "Tag Inválida")
                RETORNAR_PARA_INICIO
            FIM_SE

            // [VALIDAÇÃO 2] Verificação cadastral do colaborador
            SE (status_funcionario != "Ativo") ENTÃO
                EXIBIR "ALERTA: Acesso negado. Funcionário inativo no sistema."
                REGISTRAR_LOG(tentativa_acesso, "Funcionário Inativo")
                RETORNAR_PARA_INICIO
            FIM_SE

            // [VALIDAÇÃO 3] Verificação de nível de privilégio/permissão
            SE (permissao_acesso != "Autorizado") ENTÃO
                EXIBIR "ALERTA: Acesso negado. Usuário sem nível de permissão necessário."
                REGISTRAR_LOG(tentativa_acesso, "Sem Permissão")
                RETORNAR_PARA_INICIO
            FIM_SE

            // --- Acesso Concedido ---
            EXIBIR "SUCESSO: Acesso liberado!"
            ACIONAR_HARDWARE(trava_porta, LIBERAR_ENTRADA)
            REGISTRAR_LOG(evento_acesso, "Entrada Confirmada")

            EXIBIR "Monitoramento: Funcionário em ambiente restrito."

            // --- Sub-rotina de Controle de Retorno (Saída) ---
            FAÇA
                SE (tag_saida_detectada) ENTÃO
                    LER tag_saida
                    EXIBIR "Aviso: Processando tag de saída..."
                FIM_SE
            ENQUANTO (tag_saida != tag_autorizada)

            // --- Fechamento do Ciclo de Acesso ---
            EXIBIR "SUCESSO: Saída autorizada e registrada."
            REGISTRAR_LOG(evento_acesso, "Saída Confirmada")

        FIM_SE

    FIM_ENQUANTO

FIM DO SISTEMA
